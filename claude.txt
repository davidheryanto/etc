# claude-trace
# Record all your interactions with Claude Code as you develop your projects
# https://github.com/badlogic/lemmy/tree/main/apps/claude-trace
# Logs are saved to .claude-trace/log-YYYY-MM-DD-HH-MM-SS.{jsonl,html}

npm install -g @mariozechner/claude-trace
claude-trace --include-all-requests
------------------------------------------------------------

# Common settings:
# cat ~/.claude/settings.json
{
  "permissions": {
    "allow": [
      "WebSearch",
      "WebFetch",
      "Bash(ls:*)",
      "Bash(find:*)",
      "Bash(git add:*)",
      "Bash(git commit:*)",
      "Bash(git push:*)"
    ],
    "deny": []
  }
}

------------------------------------------------------------
# Multi-line prompts

# Option 1: Heredoc (inline, use 'EOF' to prevent $variable expansion)
claude <<'EOF'
Your multi-line
prompt here
EOF

# Option 2: From file
claude < prompt.txt

# Option 3: Pipe
cat prompt.txt | claude

# Option 4: Save prompt to file first, then run
cat > /tmp/prompt.txt <<'EOF'
@src/file1.py
@src/file2.py

Your instructions here...
EOF
claude < /tmp/prompt.txt
