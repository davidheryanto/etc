# Disable persistence
# https://stackoverflow.com/questions/28785383/how-to-disable-persistence-with-redis
redis-server --save "" --appendonly no

# Disable Transparent Huge Pages (THP) support
# http://antirez.com/news/84
sudo sh -c 'echo never > /sys/kernel/mm/transparent_hugepage/enabled'

# Common commands
redis-cli -h HOSTNAME -p PORT ping
redis-cli flushall
redis-cli incr mycounter
redis-cli lpush mylist a b c d

# Configure slave in redis.conf
slaveof [REDIS_HOST] 6379
slave-read-only yes

# Monitor redis with telnet
# https://stackoverflow.com/questions/21795340/linux-install-redis-cli-only
telnet localhost 6379

monitor
