# az storage: suppress warnings when piping JSON output
# "no credentials provided" warning prints to stdout, breaking JSON parsing
# Fix: use --only-show-errors
az storage blob list --only-show-errors --account-name X --container-name Y --output json | python3 ...

# Alternative: provide explicit auth to avoid the warning entirely
az storage blob list --auth-mode login ...
export AZURE_STORAGE_AUTH_MODE=login
