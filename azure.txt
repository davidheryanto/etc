# Install azure-cli on Fedora
sudo dnf install azure-cli

# If az crashes (e.g. 'NoneType' object is not iterable),
# miniconda's azure-* packages may conflict. Clean them:
pip uninstall azure-cli azure-cli-core azure-nspkg azure-cli-nspkg \
    azure-storage-blob azure-storage-common azure-multiapi-storage -y

# Scripting with az storage
# - Use --only-show-errors to suppress credential warnings from stdout
# - Always save to file first, then parse (large output breaks direct piping)
az storage blob list --only-show-errors --account-name X -c Y --output json > /tmp/blobs.json
python3 -c "import json; blobs = json.load(open('/tmp/blobs.json')); ..."
